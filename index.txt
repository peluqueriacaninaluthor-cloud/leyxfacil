      _extractTopic(query) {
        // Intentar extraer el tema de la consulta
        const patterns = [
          /sobre\s+([^\.\?,]+)/i,
          /de\s+([^\.\?,]+)/i,
          /para\s+([^\.\?,]+)/i,
          /acerca\s+de\s+([^\.\?,]+)/i,
          /tema\s+([^\.\?,]+)/i
        ];

        for (const pattern of patterns) {
          const match = query.match(pattern);
          if (match) {
            const topic = match[1].trim();
            if (topic.length > 3 && topic.length < 100) {
              return topic;
            }
          }
        }

        // Si no se detecta, tomar palabras clave
        const words = query.split(' ');
        const legalKeywords = ['derecho', 'ley', 'constitución', 'código', 'sentencia', 'tutela', 'contrato', 'delito',
          'familia', 'laboral', 'penal', 'civil', 'administrativo'
        ];
        const found = words.filter(w => legalKeywords.some(k => w.toLowerCase().includes(k)));

        return found.length > 0 ? found.slice(0, 3).join(' ') : null;
      },

      _handleFiles(files) {
        for (const file of files) {
          // Validar tamaño (máx 10MB)
          if (file.size > 10 * 1024 * 1024) {
            Toast.warning(`⚠️ El archivo "${file.name}" supera el límite de 10MB`, 4000);
            continue;
          }

          // Validar formato
          const validTypes = ['.txt', '.pdf', '.docx', '.json'];
          const ext = '.' + file.name.split('.').pop().toLowerCase();
          if (!validTypes.includes(ext)) {
            Toast.warning(`⚠️ Formato no soportado: ${ext}`, 3000);
            continue;
          }

          // Leer contenido para archivos de texto
          const reader = new FileReader();
          reader.onload = (e) => {
            let content = null;
            if (ext === '.txt' || ext === '.json') {
              try {
                content = e.target.result;
              } catch (err) { /* ignore */ }
            }

            AppState.files.push({
              id: Helpers.generateId(),
              name: file.name,
              size: file.size,
              type: ext,
              uploaded: new Date().toISOString(),
              content: content || null
            });

            Render.files();
            Toast.success(`📤 Archivo subido: ${file.name}`, 3000);
          };

          if (ext === '.txt' || ext === '.json') {
            reader.readAsText(file);
          } else {
            // Para PDF y DOCX solo guardamos metadata
            AppState.files.push({
              id: Helpers.generateId(),
              name: file.name,
              size: file.size,
              type: ext,
              uploaded: new Date().toISOString(),
              content: null
            });
            Render.files();
            Toast.success(`📤 Archivo subido: ${file.name}`, 3000);
          }
        }
      }
    };

          if (e.target === overlay) this.close();
        });

        document.querySelectorAll('.mobile-link').forEach(link => {
          link.addEventListener('click', (e) => {
            this.close();
            const targetId = link.getAttribute('href');
            if (targetId && targetId.startsWith('#')) {
              e.preventDefault();
              const target = document.querySelector(targetId);
              if (target) {
                const offset = header?.offsetHeight || 72;
                const pos = target.getBoundingClientRect().top + window.pageYOffset - offset - 20;
                window.scrollTo({ top: pos, behavior: 'smooth' });
              }
            }
          });
        });

        // Scroll effect
        let lastScroll = 0;
        window.addEventListener('scroll', Helpers.debounce(() => {
          const current = window.pageYOffset || document.documentElement.scrollTop;
          if (header) {
            header.classList.toggle('scrolled', current > 20);
          }
          lastScroll = current;
        }, 50), { passive: true });

        // Footer año
        document.getElementById('year').textContent = new Date().getFullYear();

        // Modales
        document.querySelectorAll('.footer-link[data-modal]').forEach(btn => {
          btn.addEventListener('click', () => {
            const modalId = `modal-${btn.dataset.modal}`;
            const modal = document.getElementById(modalId);
            if (modal) modal.style.display = 'flex';
          });
        });

        document.querySelectorAll('.modal-close-btn, .modal-overlay .modal-close-btn').forEach(btn => {
          btn.addEventListener('click', () => {
            const modal = btn.closest('.modal-overlay');
            if (modal) modal.style.display = 'none';
          });
        });

        document.querySelectorAll('.modal-overlay').forEach(modal => {
          modal.addEventListener('click', (e) => {
            if (e.target === modal) modal.style.display = 'none';
          });
        });

        document.addEventListener('keydown', (e) => {
          if (e.key === 'Escape') {
            document.querySelectorAll('.modal-overlay').forEach(m => m.style.display = 'none');
          }
        });
      },

        if (this._isOpen) this.close();
        else this.open();
      },

        const hamburger = document.getElementById('hamburgerBtn');
        const overlay = document.getElementById('mobileNavOverlay');

        const items = AppState.materials[type] || [];
        const item = items[index];
        if (item) {
          const blob = new Blob([item.content], { type: 'text/plain' });
          const url = URL.createObjectURL(blob);
          const a = document.createElement('a');
          a.href = url;
          a.download = `${type}-${Date.now()}.txt`;
          a.click();
          URL.revokeObjectURL(url);
          Toast.success('📥 Material exportado', 2000);
        }
      },

      deleteMaterial(index, type) {
        if (confirm('¿Eliminar este material?')) {
          AppState.materials[type].splice(index, 1);
          Render.materials(type);
          Toast.success('🗑️ Material eliminado', 2000);
        }
      },

      addMaterial(type, title, content) {
        if (!AppState.materials[type]) AppState.materials[type] = [];
        AppState.materials[type].push({
          id: Helpers.generateId(),
          title: title || `${type} ${AppState.materials[type].length + 1}`,
          content: content,
          created: new Date().toISOString()
        });
        Render.materials(type);
        Toast.success(`✅ ${type} generado correctamente`, 3000);
      }
